与Git分支操作相关的命令

2016-09-26
工具 

# 查看当前分支, *代表的分支为当前分支
git branch 

# 查看所有分支
git branch -a

# 查看远程分支

# 创建Hello分支
git branch Hello

# 切换到Hello分支
git checkout Hello

# 将Hello分支推送到远程仓库
git add .
git commit -m 'create Branch Hello'
git push origin master

# 删除Hello分支
git branch -d Hello

# 强制删除Hello分支
git branch -D Hello

# 查看冲突文件
git diff

# 合并分支， master为主分支，Hello为临时分支，将Hello分支上的修改合并到master分支上
git fetch origin Hello

# 查看不同文件，决定是否要合并
git diff Hello

# 合并分支
git merge  Hello

# 撤销合并
git --reset --hard HEAD 

# 或者你已经把合并后的代码提交，但还是想把他们撤销
git --reset --hard ORIG_HEAD
