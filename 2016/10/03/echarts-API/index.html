标题组件title

标题组件, 包含主标题和副标题.

在Echarts2.X中单个图表中只能包含一个标题组件, 但是在Echarts3中可以存在多个标题组件, 这在需要标题进行排版, 或者单个图标中的多个图标都需要标题是会比较有用.

var myCharts = echarts.init(document.getElementById('myCharts'));
myCharts.setOption({
    title: {
        show: true, // 是否显示标题组件, 默认为true
        text: '主标题', // 主标题
        link: '主标题链接', // 默认为空
        target: 'self', // 指定窗口打开链接, self当前窗口打开, blank新窗口打开, 默认blank
        textStyle: { // 标题文字样式
            color: '#f00', // 颜色, 默认#333
            fontStyle: 'italic', // 风格, 默认normal, 可选: normal/italic/oblique
            fontWeight: 'bold', // 粗细, 默认normal, 可选: normal/bold/bolder/lighter/100|200|300|400...
            fontFamily: 'sans-serif', // 文字系列, 默认sans-serif
            fontSize: 18 // 大小, 默认18
        },
        textAlign: 'center', // 标题文本水平对齐方式, 默认根据标题位置决定, 可选值: left/center/right
        textBaseline: 'middle', // 标题文本垂直对齐方式, 默认根据标题位置决定, 可选值: top/middle/bottom
        subtext: '副标题', // 副标题
        sublink: '副标题链接', // 副标题链接
        subtarget: 'self', // 副标题窗口打开方式, 默认blank
        subtextStyle: {
            ... // 和textStyle参数一致
        },
        itemGap: 10, // 主副标题之间的间距, 默认10
        zlevel: 0, // 所有图形的zlevel值, 用于canvas分层, zlevel 大的 Canvas 会放在 zlevel 小的 Canvas 的上面。
        z: 2, // 组件所有图形的z值, 控制图形的先后顺序, z值小的图形会被z值大的图形覆盖
        left: 'auto', // 组件距离容器左侧的距离, 默认值为auto, 值可以具体的像素值, 也可以为百分比, 也可以为left/center/right
        top: 'auto', // 组件距离容器顶部的距离
        right: 'auto', // 组件距离容器右侧的距离
        bottom: 'auto', // 组件距离容器底部的距离
        backgroundColor: 'transparent', // 背景色, 默认透明, 颜色可以使用RGBA表示
        borderColor: '#ccc', // 边框颜色, 默认为#ccc
        borderWidth: 0, // 边框粗细
        shadowBlur: 10, // 图形阴影的模糊大小, 该属性配合 shadowColor,shadowOffsetX, shadowOffsetY 一起设置图形的阴影效果。注意：此配置项生效的前提是，设置了 show: true 以及值不为 tranparent 的背景色 backgroundColor
        shadowColor: '#ccc', // 阴影颜色
        shadowOffsetX: 0, // 阴影水平方向上的偏移距离.此配置项生效的前提是,设置了 show: true。
        shadowOffsetY: 0 // 阴影垂直方向上的偏移距离.此配置项生效的前提是,设置了 show: true。
    }
});

图例组件legend

图例组件展现了不同系列的标记(symbol)，颜色和名字。可以通过点击图例控制哪些系列不显示。

var myCharts = echarts.init(document.getElementById('myCharts'));
myCharts.setOption({
    show: true, // 是否显示
    zlevel: 0, // 所有图形的zlevel值
    z: 2, // 左右图形的z值
    left: 'auto', // 组件距离容器顶部的距离
    top: 'auto',
    right: 'auto',
    bottom: 'auto',
    width: 'auto',
    height: 'auto',
    orient: 'auto',
    aligh: 'auto',
    padding: 5,
    itemGap: 10,
    itemWidth: 255,
    itemHeight: 14,
    formatter: null, // 图例文本的内容格式器，支持字符串模板和回调函数两种形式。默认为null
    selectedMode: true, // 图例选择的模式, 默认开启图例选择, 可以设置成false关闭
    inacticeColor: '#ccc', //图例关闭时的颜色
    selected: { // 图例选中状态表
        '系列1': true, // 选中
        '系列2': false // 不选中
    },
    textStyle: { // 组件公用的文本样式

    },
    tooltip: { // 默认不显示
        show: true
    },
    data: [{ // 组件的数据数组

    }],
    backgroundColor: 'transparent', // 背景色, 默认透明
    borderColor: '#ccc', // 边框颜色
    borderWidth: 1, // 边框粗细
    shadowBlur: 1, // 阴影模糊大小
    shadowColor: '#f00', // 阴影模糊颜色
    shadowOffsetX: 1, // 阴影水平方向上的偏移距离
    shadowOffsetY: 1 // 阴影垂直方向上的偏移距离
});
